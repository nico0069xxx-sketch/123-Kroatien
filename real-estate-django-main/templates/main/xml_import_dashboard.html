<?xml version="1.0" encoding="UTF-8"?>
<properties>
  <property>
    <title>Villa am Meer</title>
    <description>Wunderschoene Villa...</description>
    <price>450000</price>
    <city>Rovinj</city>
    <type>villa</type>
    <status>sale</status>
    <bedrooms>4</bedrooms>
    <bathrooms>3</bathrooms>
    <size>250</size>
    <image>https://...</image>
  </property>
</properties>

Unterstuetzte Felder: title/naziv, description/opis, price/cijena, city/grad, type/tip, status/stanje, bedrooms/sobe, bathrooms/kupatila, size/povrsina, image/slika
